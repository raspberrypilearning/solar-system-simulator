def mouse_pressed():
# Put code to run when the mouse is pressed here
  pixel_colour = color(get(mouse_x, mouse_y))

  if pixel_colour == mercury['colour']:
    print(mercury['name'])
    print(mercury['info'])
  elif pixel_colour == venus['colour']:
    print(venus['name'])
    print(venus['info'])
  elif pixel_colour == earth['colour']:
    print(earth['name'])
    print(earth['info'])
